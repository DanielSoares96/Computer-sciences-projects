CREATE TABLE cliente (
Nome varchar(400) UNIQUE,
Login varchar(400) UNIQUE,
Morada varchar(400),
Pass_word varchar(400),
DataAdesao date,
CartaoCredito integer ,
Validade varchar(30),
Primary key (Login)
);


CREATE TABLE pagamento (
LoginPagamento varchar(400),
MesPagamento varchar(400),
DataPagamento date,
Mensalidade float(3),
Foreign Key (LoginPagamento) references cliente(Login)
);


CREATE TABLE filme (
DataEstreia date,
Lingua varchar(400),
URL varchar(1000) UNIQUE,
Path varchar(1000),
Resumo varchar(5000),
Primary key (URL)
);


CREATE TABLE filmeLocal(
LocalFilmagem varchar(1000),
URLLocal VARCHAR(400),
Primary key (LocalFilmagem,URLLocal),
Foreign key (URLLocal) references filme(URL)
);


CREATE TABLE filmetitulo(
URLTitulo VARCHAR(100),
TituloTitulo varchar(400) UNIQUE,
Linguagem varchar(400),
Primary key (URLTitulo,TituloTitulo),
Foreign key (URLTitulo) references filme(URL)
);


CREATE TABLE filmeGenero(
URLGenero varchar(400),
Genero varchar(400),
Primary key (URLGenero,Genero),
Foreign key (URLGenero) references filme(URL)
);


CREATE TABLE filmePremio(
URLPremio varchar(400),
PremiosF varchar(1000),
Primary key (URLPremio,PremiosF),
Foreign key (URLPremio) references filme(URL)
);


CREATE TABLE pessoal (
NomePessoal varchar(400) UNIQUE,
Nacionalidade varchar(400),
DataNascimento date,
PRIMARY KEY (NomePessoal,DataNascimento)
);


CREATE TABLE historialF(
LoginHistorial VARCHAR(400),
URLHistorial VARCHAR(400),
DataVisual date,
Primary key (LoginHistorial,URLHistorial),
Foreign Key (LoginHistorial) references cliente(Login),
Foreign key (URLHistorial) references filme(Url)
);


CREATE TABLE actor(
NomeAtor VARCHAR(400),
URLAtor VARCHAR(400),
Primary key (NomeAtor,URLAtor),
Foreign key (URLAtor) references filme(URL)
);


CREATE TABLE director(
NomeDiretor VARCHAR(400),
URLDiretor VARCHAR(100),
Primary key (NomeDiretor,URLDiretor),
Foreign Key (NomeDiretor) references pessoal(NomePessoal),
Foreign key (URLDiretor) references filme(Url)
);


CREATE TABLE directorPremio(
NomeDiretor VARCHAR(400),
PremioDiretor VARCHAR(400),
TituloDiretor VARCHAR(400),
URLDiretor VARCHAR(100),
Primary key (NomeDiretor,URLDiretor,PremioDiretor),
Foreign Key (NomeDiretor) references pessoal(NomePessoal)
);


CREATE TABLE actorPremio(
NomeAtor VARCHAR(400),
TituloAtor VARCHAR(400),
URLAtor VARCHAR(100),
PremioAtor VARCHAR(400),
Primary key (NomeAtor,URLAtor,PremioAtor),
Foreign Key (NomeAtor) references pessoal(NomePessoal)
);
